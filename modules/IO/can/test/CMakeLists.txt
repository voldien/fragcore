
FIND_PACKAGE(GTest REQUIRED)
IF(GTEST_FOUND)
	MESSAGE(STATUS "Gtest found: ${GTEST_BOTH_LIBRARIES}")
ELSE()
	MESSAGE(ERROR "Gtest is required in order to compile and run the tests")
ENDIF()

# Source files.
FILE(GLOB CANTEST_SOURCE_FILES 	${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)
FILE(GLOB CANTEST_HEADER_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.h)

ADD_EXECUTABLE(can-io-test ${CANTEST_SOURCE_FILES} ${CANTEST_HEADER_FILES})
TARGET_LINK_LIBRARIES(can-io-test GTest::GTest GTest::Main fragcore-ionet-can )
TARGET_INCLUDE_DIRECTORIES(can-io-test PRIVATE ${GTEST_INCLUDE_DIRS} ${CMAKE_CURRENT_SOURCE_DIR}/)
ADD_DEPENDENCIES(can-io-test fragcore-ionet-can)

ADD_TEST(NAME can-io-test
		COMMAND can-io-test
		WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
